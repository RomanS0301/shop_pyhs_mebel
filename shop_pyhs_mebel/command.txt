python -m venv env\myshop  создание виртуальной среды проекта
.\env\myshop\Scripts\activate  активация виртуальной среды


django-admin startproject myshop старт проекта
django-admin startapp shop создание приложения

pip install django==4.2.23 установка нужной версии django
pip install Pillow
pip install celery==5.5.3
pip install flower

docker pull rabbitmq установка RabbitMQ c Dockera
docker run -it --rm --name rabbitmq -p 5672:5672 -p 15672:15672 rabbitmq:management запуск сервера
celery -A myshop worker -l info запуск работника celery
celery -A myshop flower
